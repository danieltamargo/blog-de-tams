```python
# Definimos la clase padre
class Animal:
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad

    def emitir_sonido(self):
        print("¡Grrrrr!")

# Y ahora podemos crear subclases que hereden de ella
class Perro(Animal):
    def __init__(self, nombre, edad):
        super().__init__(nombre, edad) # A través de super() llamamos al constructor de la clase padre

    def emitir_sonido(self):
        print("¡Guau guau!")

class Gato(Animal):
    def __init__(self, nombre, edad):
        super().__init__(nombre, edad)

    def emitir_sonido(self):
        print("¡Miau!")
```
